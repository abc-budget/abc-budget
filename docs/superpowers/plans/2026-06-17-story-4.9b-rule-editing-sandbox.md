**Goal:** Add rule EDITING (edit/delete/reorder) + the FEAT-029 SANDBOX (virtual decision tree, diff preview,
apply/cancel) to the merged 4.9a S3c categorization UI, by wiring the merged 4.5 `RuleSandboxSession` over a new
v5 EngineClient surface.

**Architecture:** The 4.5 `RuleSandboxSession` (current/virtual/engaged, `classify`/`submit`/`computeDiff`/
`apply`/`cancel`) is the single edit funnel. It is held WORKER-SIDE in a `Map<sessionId, RuleSandboxSession>`
on `CategorizationServiceImpl` (the `SessionRegistry` precedent), retained only while engaged. Five new v5 wire
methods drive it; `importCategorizedRows` becomes sandbox-aware (categorizes against the virtual tree + sets
`previousCategoryId` on changed rows → the 4.9a `CatChip.previous` old→new seam). The React side adds a
`SandboxBar`, edit/delete/reorder affordances on the rules list, a dynamic save button, and the orange-frame
engaged state.

**Tech Stack:** TypeScript; pnpm workspace (`packages/engine` + `apps/web`); React 18 + vitest + RTL (jsdom);
Playwright for the 390px live-measure; the engine's Web-Worker EngineClient wire (versioned contract).

```ts
// Discriminated union MIRRORING the engine EditAction (rule-sandbox.ts:57-62), but serializable:
//   - category → categoryId (resolved worker-side via categoriesById)
//   - Rule[]   → ConditionDTO[] (mapped worker-side via conditionsToRules)
//   - ComplexRule → { conditions, categoryId }
export type EditActionDTO =
  | { kind: 'categoryOnly'; ruleId: number; categoryId: string }
  | { kind: 'appendEnd'; conditions: ConditionDTO[]; categoryId: string }
  | { kind: 'editConditions'; ruleId: number; before: ConditionDTO[]; after: ConditionDTO[] }
  | { kind: 'reorder'; order: number[] }   // ruleIds in the NEW order
  | { kind: 'delete'; ruleId: number };

export interface SandboxStateDTO {
  readonly engaged: boolean;
  readonly count: number;   // computeDiff().length — the «N операцій» banner count
}
```
New `EngineClient` method signatures (engine-client.ts):
```ts
rulesClassify(sessionId: string, action: EditActionDTO): Promise<'live' | 'sandbox'>;
rulesSubmitEdit(sessionId: string, action: EditActionDTO): Promise<SandboxStateDTO>;
sandboxState(sessionId: string): Promise<SandboxStateDTO>;
sandboxApply(sessionId: string): Promise<void>;
sandboxCancel(sessionId: string): Promise<void>;
// MODIFIED — opts gains changedOnly:
importCategorizedRows(
  sessionId: string,
  opts: { offset: number; count: number; segment: 'all' | 'uncat'; draft?: ConditionDTO[]; changedOnly?: boolean },
): Promise<CategorizedWindowDTO>;
```
Worker-side semantics:
- `rulesClassify` — PURE preview (drives the dynamic button). No session retained. Returns `classifyEditAction(action)`.
- `rulesSubmitEdit` — the funnel. Get-or-create the session; `await session.submit(action)`; if `session.engaged`
  → **retain** in the map, return `{ engaged: true, count: computeDiff().length }`; else (live persisted, or a
  canonical no-op) → **delete** from the map, return `{ engaged: false, count: 0 }`.
- `sandboxState` — `{ engaged: session?.engaged ?? false, count: session?.engaged ? computeDiff().length : 0 }`.
  (Resume + banner refresh.)
- `sandboxApply` — `await session.apply()` (`saveDecisionTree` delta) → delete from the map.
- `sandboxCancel` — `session?.cancel()` → delete from the map.
- `importCategorizedRows` — when `sandboxes.get(sessionId)?.engaged`: overlay `computeDiff()` onto the window
  (changed rows get `categoryId = newCategoryId`, `previousCategoryId = oldCategoryId`); `changedOnly` filters
  the passing list to the diff rows. NOT engaged → EXACTLY 4.9a behavior (no `previousCategoryId`, `changedOnly`
  ignored).
- `importAbort` drops the sandbox (`categorization.dropSandbox(sessionId)`).

```ts
// field:
private readonly sandboxes = new Map<string, RuleSandboxSession>();

// ── helpers ──────────────────────────────────────────────────────────────────
private async buildCategoriesById(): Promise<Map<string, Category>> {
  const byId = new Map<string, Category>();
  for (const c of await this.categoriesService.list({ includeArchived: true })) {
    if (c.id !== undefined) byId.set(c.id, c);
  }
  return byId;
}

private async buildSandboxDeps(sessionId: string): Promise<RuleSandboxDeps> {
  const rows = await this.getSessionRows(sessionId);
  const currentTree = await this.rulePersistence.reload();
  const periods = distinctPeriods(rows);                       // the same helper autoCategorize uses
  const { overrideMap, categoriesById } = await loadOverrideMap(this.footprintDao, this.categoriesService, periods);
  return { importRows: rows, overrideMap, categoriesById, currentTree, persistence: this.rulePersistence };
}

private toEditAction(dto: EditActionDTO, categoriesById: Map<string, Category>): EditAction {
  switch (dto.kind) {
    case 'reorder':
      return { kind: 'reorder', order: dto.order };           // ruleIds in new order — see Step 8 reorder pin
    case 'delete':
      return { kind: 'delete', ruleId: dto.ruleId };
    case 'editConditions':
      return { kind: 'editConditions', ruleId: dto.ruleId,
        before: this.conditionsToRules(dto.before), after: this.conditionsToRules(dto.after) };
    case 'categoryOnly': {
      const category = categoriesById.get(dto.categoryId);
      if (!category) throw new TypeError(`[abc-engine] categoryOnly: unknown categoryId '${dto.categoryId}'`);
      return { kind: 'categoryOnly', ruleId: dto.ruleId, category };
    }
    case 'appendEnd': {
      const category = categoriesById.get(dto.categoryId);
      if (!category) throw new TypeError(`[abc-engine] appendEnd: unknown categoryId '${dto.categoryId}'`);
      return { kind: 'appendEnd', rule: new ComplexRuleImpl(this.conditionsToRules(dto.conditions), category) };
    }
  }
}

async rulesClassify(_sessionId: string, dto: EditActionDTO): Promise<'live' | 'sandbox'> {
  const categoriesById = await this.buildCategoriesById();
  return classifyEditAction(this.toEditAction(dto, categoriesById));
}

  let categoriesById: Map<string, Category>;
  if (session) {
    categoriesById = await this.buildCategoriesById();        // refetch — a category may have been created mid-sandbox
  } else {
    const deps = await this.buildSandboxDeps(sessionId);
    categoriesById = deps.categoriesById;
    session = new RuleSandboxSession(deps);
  }
  await session.submit(this.toEditAction(dto, categoriesById));
  if (session.engaged) {
    this.sandboxes.set(sessionId, session);
    return { engaged: true, count: session.computeDiff().length };
  }
  this.sandboxes.delete(sessionId);                            // live persisted, or canonical no-op
  return { engaged: false, count: 0 };
}

sandboxState(sessionId: string): SandboxStateDTO {
  const session = this.sandboxes.get(sessionId);
  return session?.engaged ? { engaged: true, count: session.computeDiff().length } : { engaged: false, count: 0 };
}

async sandboxApply(sessionId: string): Promise<void> {
  const session = this.sandboxes.get(sessionId);
  if (session?.engaged) await session.apply();
  this.sandboxes.delete(sessionId);
}

sandboxCancel(sessionId: string): void {
  this.sandboxes.get(sessionId)?.cancel();
  this.sandboxes.delete(sessionId);
}

dropSandbox(sessionId: string): void {
  this.sandboxes.delete(sessionId);
}
```
