# Story 1.1 — PWA scaffold + UI/Engine skeleton — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship an installable, offline-capable React+TS+Vite PWA whose UI talks to a framework-free engine **only** through an async `EngineClient`, with the UI/Engine boundary enforced structurally (package `exports` + TS project references + ESLint) and the worker-relocation seam *verified* by tests.

**Architecture:** A pnpm workspace with two packages — `@abc-budget/engine` (plain TS; public surface = `EngineClient` interface + DTOs + factories; impl is internal) and `@abc-budget/web` (React PWA). The UI holds an `EngineClient` produced by a factory; `createDirectEngineClient()` runs the engine in-thread today, `createWorkerEngineClient(worker)` (spike/test only) proves the same interface survives a `postMessage` thread hop — so moving the engine into a worker is zero-UI-change. Offline via `vite-plugin-pwa` `generateSW`. Firebase Hosting config is authored and offline is proven locally; the live deploy is left to the founder.

**Tech Stack:** Node 24, pnpm 11, TypeScript 5.7 (project references, `moduleResolution: bundler`), Vite 6 + `@vitejs/plugin-react`, `vite-plugin-pwa` (Workbox `generateSW`), React 19, Vitest 3 + `@vitest/web-worker`, ESLint 9 flat config + `typescript-eslint`, Firebase Hosting (`npx firebase-tools`).

**Source spec:** `docs/superpowers/specs/2026-06-09-story-1.1-pwa-engine-skeleton-design.md`
**Branch:** `story/1.1-pwa-engine-skeleton` (already created; the spec commit is its first commit).

import type { EngineRequest, EngineResponse } from '../client/protocol';
import { createPingEngine } from './ping-engine';

// In a worker, `self` is the worker global scope. We type only what we use, so the
// package can compile under the DOM lib without pulling in the WebWorker lib.
const ctx = self as unknown as {
  onmessage: ((event: MessageEvent<EngineRequest>) => void) | null;
  postMessage: (data: EngineResponse) => void;
};

const engine = createPingEngine();

ctx.onmessage = async (event) => {
  const { id, method, args } = event.data;
  try {
    const fn = engine[method] as (...a: unknown[]) => Promise<unknown>;
    const value = await fn(...args);
    ctx.postMessage({ id, ok: true, value });
  } catch (error) {
    ctx.postMessage({ id, ok: false, error: String(error) });
  }
};
```

import type { EngineClient, EngineVersion } from './engine-client';
import type { EngineMethod, EngineRequest, EngineResponse } from './protocol';

interface Pending {
  resolve: (value: unknown) => void;
  reject: (reason: unknown) => void;
}

/**
 * Builds an EngineClient backed by a Worker over postMessage. Spike/test only — not
 * exported from the package barrel. Proves the EngineClient contract is identical whether
 * the engine runs in-thread or in a worker (NFR-003 relocation seam).
 */
export function createWorkerEngineClient(worker: Worker): EngineClient {
  let nextId = 1;
  const pending = new Map<number, Pending>();

  worker.onmessage = (event: MessageEvent<EngineResponse>) => {
    const { id, ok, value, error } = event.data;
    const entry = pending.get(id);
    if (!entry) return;
    pending.delete(id);
    if (ok) entry.resolve(value);
    else entry.reject(new Error(error ?? 'engine worker error'));
  };

  const call = (method: EngineMethod, args: unknown[]): Promise<unknown> => {
    const id = nextId++;
    const request: EngineRequest = { id, method, args };
    return new Promise<unknown>((resolve, reject) => {
      pending.set(id, { resolve, reject });
      worker.postMessage(request);
    });
  };

  return {
    ping: (message) => call('ping', [message]) as Promise<string>,
    getVersion: () => call('getVersion', []) as Promise<EngineVersion>,
  };
}
```

```js
import { deflateSync } from 'node:zlib';
import { mkdirSync, writeFileSync } from 'node:fs';

function crc32(buf) {
  let c = ~0;
  for (let i = 0; i < buf.length; i++) {
    c ^= buf[i];
    for (let k = 0; k < 8; k++) c = (c >>> 1) ^ (0xedb88320 & -(c & 1));
  }
  return ~c >>> 0;
}

function chunk(type, data) {
  const len = Buffer.alloc(4);
  len.writeUInt32BE(data.length, 0);
  const body = Buffer.concat([Buffer.from(type, 'latin1'), data]);
  const crc = Buffer.alloc(4);
  crc.writeUInt32BE(crc32(body), 0);
  return Buffer.concat([len, body, crc]);
}

function png(size, [r, g, b]) {
  const sig = Buffer.from([137, 80, 78, 71, 13, 10, 26, 10]);
  const ihdr = Buffer.alloc(13);
  ihdr.writeUInt32BE(size, 0);
  ihdr.writeUInt32BE(size, 4);
  ihdr[8] = 8; // bit depth
  ihdr[9] = 2; // color type: truecolor
  const row = Buffer.alloc(1 + size * 3);
  for (let x = 0; x < size; x++) {
    row[1 + x * 3] = r;
    row[1 + x * 3 + 1] = g;
    row[1 + x * 3 + 2] = b;
  }
  const raw = Buffer.concat(Array.from({ length: size }, () => row));
  const idat = deflateSync(raw);
  return Buffer.concat([
    sig,
    chunk('IHDR', ihdr),
    chunk('IDAT', idat),
    chunk('IEND', Buffer.alloc(0)),
  ]);
}

const teal = [13, 115, 119];
mkdirSync('apps/web/public', { recursive: true });
writeFileSync('apps/web/public/pwa-192x192.png', png(192, teal));
writeFileSync('apps/web/public/pwa-512x512.png', png(512, teal));
console.log('PWA icons written to apps/web/public/');
```
